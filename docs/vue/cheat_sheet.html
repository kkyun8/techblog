# Javascript Event

# コンポーネントから keydown イベント →→→ addEventListener を実装したい場合

  created() {
    document.addEventListener("keydown", this.methodName);
  },
  // 必ずremove、をセット
  beforeDestroy () {
    document.removeEventListener("keydown", this.methodName);
  }
